        \newcommand{\sect}[2] {
            \addtocounter{section}{1}

        }

            \addtocounter{subsection}{1}
            \subsection*{\thesubsection.\,#1}
            \addcontentsline{toc}{subsection}{ \texorpdfstring{\quad \thesubsection.\qquad\ #2}{Lg}}
        }

            \subsubsection*{\thesubsubsection.\,#1}

        }
